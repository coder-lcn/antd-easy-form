## Install

```bash
npm install --save antd-easy-form
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'antd-easy-form'
import 'antd-easy-form/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
